/* 计算一个 Canvas 中有笔画的区域 */

function calcArea(
    view: Uint32Array,
    width: number,
    height: number
): {
    top: number;
    right: number;
    bottom: number;
    left: number;
} {
    let top, right, bottom, left;
    let i, j, k, l;
    for (i = 0; i < view.length; i++) {
        if (view[i]) {
            top = Math.floor(i / width);
            left = i % width;
            right = left;
            bottom = top;
            break;
        }
    }
    for (j = view.length - 1; j > i; j--) {
        if (view[j]) {
            bottom = Math.floor(j / width);
            left = Math.min(j % width, left);
            right = Math.max(j % width, right);
            break;
        }
    }
    for (let tmp = 0; tmp < left; tmp++) {
        for (k = width * top + tmp; k < j; k += width) {
            if (view[k]) {
                left = tmp;
                break;
            }
        }
    }
    for (let tmp = 1; tmp < width - right; tmp++) {
        for (l = width * top - tmp; l < j; l += width) {
            if (view[l]) {
                right = width - tmp;
                break;
            }
        }
    }
    return {
        top,
        right,
        bottom,
        left
    };
}
