function getFiles(dir, sortKey = 'mtime') {
    let files = fs.readdirSync(dir);
    return files.sort((a, b) => {
        return (
            fs.statSync(path.join(dir, a))[sortKey] <
            fs.statSync(path.join(dir, b))[sortKey]
        );
    });
}
